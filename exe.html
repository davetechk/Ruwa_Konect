Open Paystack const payResult = await initializePaystack(orderId); orderState.paymentRef = payResult.reference; orderState.status = 'paid'; // 2. Show success showSuccessModal(orderId); showToast('success', 'Order placed!', 'Your water order is confirmed and in the queue.'); } catch (err) { if (err.message === 'Payment window closed by user') { showToast('info', 'Payment cancelled', 'You closed the payment window.
